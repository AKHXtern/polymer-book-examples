	// selecting the template element with querySelector()
	var tmpl = document.querySelector('.superman');
	//getting the <template> content
	var content = tmpl.content;
	// making some changes in the content
	content.querySelector('.animated_superman').width = 200;

	// appending the template to the body
	document.body.appendChild(content);
</script>

<!-- Creating Game of Thrones character tag -->
<!-- You can use names like (cersei-lannister, robb-stark, jon-snow, jamie-lannister, tyrion-lannister, daenerys-targaryen) -->
<show-got id="showGot" character="robb-stark"></show-got>

<script>
var character = Object.create(HTMLElement.prototype);
character.show = function(){
    // Creating info text
    var info = document.createElement('p');
    info.innerHTML = 'Hover me to feel the GOT!';
    var person = this.getAttribute('character');
    img = document.createElement('img');
    img.src = 'https://s3-us-west-2.amazonaws.com/s.cdpn.io/4273/' + person + '.png';
    img.width = 300;
    img.height = 240;
    this.appendChild(img);
    this.appendChild(info);
    
    this.onmouseover = function(){
        this.audio.play();
        info.innerHTML = ":) " + person;        
    };
    this.onmouseout = function(){
        this.audio.pause();
        info.innerHTML = ":( " + person;
    };
};
character.audio = new Audio('https://upload.wikimedia.org/wikipedia/en/a/a8/Game_of_Thrones_Main_Title_sample.ogg');


document.registerElement('show-got', {prototype: character});
showGot.show();
</script>

Hey, guys! 
<style>
  /* Style for: unresolved element (prior to registration) */
  say-hello:unresolved {
    color: white;
    background: #cc0000;
  }
  say-hello {
    transition: all 4s;
  }
</style>
<say-hello id="hello">Hey, guys!</say-hello>
<script>
  // Registration will take place in 3 seconds
  setTimeout(function() {
    document.registerElement("say-hello", {
      prototype: {
        __proto__: HTMLElement.prototype,
        sayHelloAgain: function() { console.log('I said HEY!'); }
      }
    });

    // the new type of elements is a method sayHelloAgain
    hello.sayHelloAgain();
  }, 3000);
</script>

<script>
  var proto = Object.create(HTMLButtonElement.prototype);
  proto.count = function() {
    this.innerHTML++;
  };	

  document.registerElement("timer-tag", {
    prototype: proto,
    extends: 'button'
  });
</script>

<button is="timer-tag" id="counter">0</button>

<script>
  setInterval(function() {
    counter.count();
  }, 1000);

  counter.onclick = function() {
    alert("Current value: " + this.innerHTML);
  };
</script>

<script>
  var proto = Object.create(HTMLElement.prototype);

  proto.count = function() {
    this.counter++;
    this.innerHTML = this.counter;
  };

  proto.createdCallback = function() {
    this.counter = 0;
  };

  proto.attachedCallback = function() {
    setInterval(this.count.bind(this), 1000);
  };

  document.registerElement("timer-tag", {
    prototype: proto
  });
</script>

<timer-tag id="timer">0</timer-tag>
